a.Demonstrate how to Perform a Port Scan using Scapy tool on target www.example.com.
Ans:Open cmd prompt(normal)
 Next,type pip install scapy  //(if not already installed)
 Then type: scapy //hit enter
 Type the following cmds below:
    
from scapy.all import *

target_ip = "scanme.nmap.org"
ports = [22, 80]
for port in ports:
    pkt = IP(dst=target_ip)/TCP(dport=port, flags="S")
    resp = sr1(pkt, timeout=1, verbose=0)
    if resp:
        if resp.haslayer(TCP) and resp[TCP].flags == "SA":
            print(f"Port {port} is open")
        elif resp.haslayer(TCP) and resp[TCP].flags == "RA":
            print(f"Port {port} is closed")
        else:
            print(f"Port {port} is filtered or unknown response")
    else:
        print(f"No response from port {port}") // output is: Port 22 is open Port 80 is open
